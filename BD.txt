drop database if exists ventasPc;

create database ventasPc;

use ventasPc;

create table cliente(
	idCliente int auto_increment,
	nombre varchar(50),
	apellido varchar(50),
	direccion varchar(50),
	telefono int,
	correo varchar(50),
	primary key(idCliente)
);

create table vendedor(
	idVendedor int auto_increment,
	nombre varchar(50),
	apellido varchar(50),
	primary key(idVendedor)
);





create table computadora(
	idComp int auto_increment,
	nombre varchar(50),
	precio double,
	ram varchar(50),
	procesador varchar(50),
	gabinete varchar(50),
	primary key(idComp)
);

create table factura(
	nFactura int auto_increment,
	precio double,
	fecha date,
	cyberMonday int,
	idVendedor int,
	idCliente int,
	idComp int,
	primary key(nFactura),
	foreign key(idVendedor) references vendedor(idVendedor),
	foreign key(idCliente) references cliente(idCliente),
	foreign key(idComp) references computadora(idComp)
);
